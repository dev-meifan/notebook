#### 深度优先

```js
let tree = [
    {
        value: 1,
        children: [
            { value: 3 },
            {
                value: 5,
                children: [
                    { value: 51 },
                    {
                        value: 52,
                        children: [
                            { value: 521 },
                            { value: 522 }
                        ]
                    },
                    {
                        value: 53,
                        children: [
                            { value: 531 }
                        ]
                    }
                ]
            },
        ]
    }
];

function deepFirstSearch (tree = [], nodeList = []) {
    tree.forEach(item => {

        nodeList.push(item.value);
        if (item.children) {
            DeepFirstSearch(item.children, nodeList);
        }
    });

    return nodeList;
}

deepFirstSearch(tree);
// [1, 3, 5, 51, 52, 521, 522, 53, 531]

```
